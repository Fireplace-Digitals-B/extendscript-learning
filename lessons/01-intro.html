Imagine you're a time traveler who knows modern JavaScript, but you've landed in the early 2000s. That's basically what using ExtendScript feels like! Here's why:

Modern JavaScript

// Modern JavaScript
const videoLayers = ['intro', 'main', 'outro'];
videoLayers.forEach(layer => {
    console.log(`Processing ${layer}...`);
});

// Using modern features
const settings = {
    duration: 5,
    effects: ['fade', 'blur']
};

const { duration, effects } = settings;

ExtendScript (Time Travel Mode)

// ExtendScript
var videoLayers = ['intro', 'main', 'outro'];
for (var i = 0; i < videoLayers.length; i++) {
    $.writeln("Processing " + videoLayers[i] + "...");
}

// No destructuring - old school way
var settings = {
    duration: 5,
    effects: ['fade', 'blur']
};

var duration = settings.duration;
var effects = settings.effects;
